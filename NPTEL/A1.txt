public class Exercise1_1 {
       public static void main(String[] args) {

       double radius= s.nextDouble();
       double perimeter;
       double area;

{
 System.out.println("please enter non zero positive number ");

else

 perimeter = 2 * Math.PI * radius;

 System.out.println(area);
