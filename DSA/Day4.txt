# Selection Sort
   -------------
#include <stdio.h>
void main()
{
int a[] = {50,40,20,10,30},i,j,temp;
printf("Before Sorting");
for(i=0;i<5;i++)
{ printf("%d\t",a[i]);
}
printf("\n");
for(i=0;i<4;i++) {
for(j=i+1;i<5;j++)  {
if(a[j]>a[i]) {
temp = a[i];
a[i] = a[j];
a[j] = temp;
}
printf("\n");
printf("After Sorting");
for(i=0;i<5;i++)
{ printf("%d\t",a[i]);
}
}

#Insertion Sort
  -----------------

#include <stdio.h>
void main()
{
int a[]={50,20,10,40,30},i,j,temp;
printf("Before Sorting \n");
for(i=0;i<5;i++)
{
printf("%d\t",a[i]);
}

for(i=1;i<=5;i++)
{
j=i-1;
temp=a[i];
while(j>=0 && a[j]>temp)
{
a[j+1]=a[j];
j--;
}
a[j+1]=temp;
}
printf("After Sorting \n");
for(i=0;i<5;i++)
{
printf("%d\t",a[i]);
}
}
