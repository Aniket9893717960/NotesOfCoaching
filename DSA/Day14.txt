#Circular Queue
 ---------------------
#include <stdio.h>
#define maxsize 10
int queue[maxsize],rear = -1,front =0;
void insert();
void del();
void display();

void main()
{
int ch;
do
{
printf("-----------------Circular Queue----------------");
printf("Press 1 for Insert in circular Queue\n");
printf("Press 2 for Delete in Circular Queue\n");
printf("Press 3 for Display in circular Queue\n");
printf("Press 4 for Exit\n");
printf("Enter your Choice\n");
scanf("%d",&ch);
switch(ch)
{
case 1: insert();
break;
case 2: del();
break;
case 3: display();
break;
case 4: break;
default:
printf("Invalid number");
}
}
while(ch!=4);
}

void insert()
{
if(rear ==(maxsize-1))
{
printf("Queue is overflow");
}
else if((rear+1)%maxsize < front%maxsize && front!=0)
{
printf("\n Enter a value");
rear++;
scanf("%d",&queue[rear%maxsize]);
}
else
{
printf("Enter a value :");
rear++;
scanf("%d",&queue[rear]);
}
}

void del()
{
if(front>rear)
{
printf("Queue is underflow");
front =0;
rear =-1;
}
else
{
printf("Deleted element is  %d:",queue[front%maxsize]);
front++;
}
}

void display()
{
int i;
if(front>rear)
{
printf("Queue is underflow");
}
else
{
for(i=front;i<=rear;i++)
{
printf("%d\t",queue[i%maxsize]);
}
}
}
