{

 sum = rev(temp);
if(sum == temp)
{
printf("Entered value is a palindrome number");
}
else
{
printf("Not a palindrome");
}
}

int rev(int temp)
{
int sum = 0,r;
while(temp!=0)
{
r= temp%10;
sum = sum*10 + r;
temp = temp/10;
}
return sum;
}
