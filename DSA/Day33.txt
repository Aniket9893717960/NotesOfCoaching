}
void hanoi(int n,char init,char final,char temp)
{
if(n==1)
{
printf("More disc 1 from %c to %c\n",init,final);
return;
}
hanoi(n-1,init,temp,final);
printf("More disc %d from %c to %c\n",n,init,final);
hanoi(n-1,temp,final,init);
}
