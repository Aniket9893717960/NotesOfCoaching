class MyThread implements Runnable
{
Thread t;
MyThread()
{
t = new Thread (this,"MyThread");
System.out.println(t);
t.start();
}
public void run(){
try{
for(int i=1;i<=5;i++){
System.out.println("MyThread"+i);
t.sleep(1000);
}
}
catch(InterruptedException e){
System.out.println(e);
}
System.out.println("Mythread exit");
}
}
