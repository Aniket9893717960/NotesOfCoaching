#Sorting of an Array
   -----------------------
i)Selection Sort
  -----------------
#include <stdio.h>

void main()
{
int a[]={50,30,10,20,40},i,j,temp;
printf("Array Before Sorting \n");
for(i=0;i<5;i++)
{
printf("%d\t",a[i]);
}
for(i=0;i<4;i++)
{
for(j=i+1;j<5;j++)
{
if(a[i]>a[j])
{
temp = a[i];
a[i] = a[j];
a[j] = temp;
}
}
}
printf("Array After Sorting \n");
for(i=0;i<5;i++)
{
printf("%d\t",a[i]);
}
}
