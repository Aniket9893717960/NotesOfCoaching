# write a Program to enter the 10 value from user and Print the reverse of the array?

#include <stdio.h>
void main()
{
int a[10],i;
printf("Enter 10 values \n");
for(i=0;i<10;i++)
{
printf("Enter value at %d position",i+1);
scanf("%d",&a[i]);
}
printf("your Entered reverse of an array :");
i--;
while(i>=0)
{
printf("%d\t",a[i]);
i--;
}
}

{
