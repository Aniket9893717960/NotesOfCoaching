Q) Print a Palindrome String with the help of Character Array?

#include <stdio.h>
#include <conio.h>
#include <string.h>

void main()
{
char a[10],*p,*q;
int i,f=0;

p=&a[0];
printf("Enter a String \n");
gets(a);

i=strlen(a)-1;

q=&a[i];
while(p<q)
{
if(*p!=*q)
{
f=1;
break;
}
p++;
q--;
}
if(f==0)
{
printf("The String is Palindrome \n");
}
else
{
printf("String is Non Palindrome \n");                                                OUTPUT       // input is MALAYALAM
}                                                                                                                     The String is Palindrome
}
