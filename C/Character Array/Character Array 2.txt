#write a Program to Enter a String from the user and Print the Length of the String?

#include <stdio.h>
#include <string.h>

void main()
{
char a[10];
int i=0;

printf("Enter a Name: \n");
gets(a);

while(a[i]!='\0')
{
i++;
}
printf("Length of the element is =%d",i);
}

                                   -----------------------------------------------------

}

#write a program to enter a String from user and print the string in reverse order?

#include <stdio.h>
#include <string.h>

void main()
{
char a[10];
int i;

printf("Enter a Name :\n");
gets(a);

printf("Reverse of String:");
for(i=(strlen(a)-1);i>=0;i--)
{
printf("%c",a[i]);
}
}
