Write  menu driven program which has following options?

#include <stdio.h>

void factorial();
void primeOrNot();
void OddorEven();

void main()
{
int ch=0;
do
{
printf("Press 1: for Factorial of number \n");
printf("Press 2: for Prime or Not \n");
printf("Press 3: for Odd or Even \n");
printf("Press 4: for Exit");
printf("Choose the value \n");
scanf("%d",&ch);
switch(ch)
{
case 1: factorial();
break;
case 2: primeOrNot();
break;
case 3: OddorEven();
break;
case 4:break;
default:
printf("Invalid position \n");
}
}
while(ch!=4);
}

void factorial()
{
int f=1,temp,n;
printf("Enter the value for Factorial \n");
scanf("%d",&n);
temp=n;
while(n>1)
{
f=f*n--;
}
printf("The factorial of %d is = %d\n",temp,f);
}

void primeOrNot()
{
int n,i,temp;
int f=0;
printf("Enter the value to Check this will be prime or Not\n");
scanf("%d",&n);
temp=n;
for(i=2;i<(n/2);i++)
{
if(n%i==0)
{
f=1;
break;
}
}
if(f==1)
{
printf("%d is not a prime number\n",temp);
}
else
{
printf("%d is a Prime number \n",temp);
}
}

void OddorEven()
{
int n,i,temp,f=0;
printf("Enter the value to Check the number is odd or an Even \n");
scanf("%d",&n);
temp=n;

while(n>0)
{
if(n%2==0)
{
f=1;
break;
}
n--;
}

if(f==1)
{
printf("%d will be an Even number \n",temp);
}
else
{
printf("%d is an Odd number \n",temp);
}
}
