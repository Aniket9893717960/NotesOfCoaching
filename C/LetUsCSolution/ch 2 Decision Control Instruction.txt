Q)Rewrite the following program using conditional operators?
a)
#include <stdio.h>
int main()
{
int x,min,max;
scanf("%d%d",&max,&x);
if(x>max)
printf("%d\t",max = x);
else
printf("%d\t",min=x);
return 0;
}
                                                   solution
#include <stdio.h>

void main()
{
int x,min,max;
scanf("%d%d",&max,&x);
x = (x>max?printf("%d",max=x):printf("%d",min=x));
}
