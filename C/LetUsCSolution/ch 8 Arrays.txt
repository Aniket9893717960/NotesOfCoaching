#include <stdio.h>
void main()
{
int a[3][3],b[3][3],c[3][3],i,j;

printf("Enter the value in first matrix \n");
for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
printf(" Enter the value at %d %d position",(i+1),(j+1));
scanf("%d",&a[i][j]);
}
}
printf("Enter value at Second matrix \n");
for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
printf("Enter the value at %d %d position",(i+1),(j+1));
scanf("%d",&b[i][j]);
}
}

for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
c[i][j]=0;

}

{
for(j=0;j<3;j++)
{

}

}
}

                                                                  Q) write a program to sort the 6*6 matrix?
                                                                                   Solution
#include <stdio.h>

void main()
{
int a[3][3],i,j,temp;
printf("Enter the value in matrix \n");
for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
printf("Enter the values at %d %d position",i+1,j+1);
scanf("%d",&a[i][j]);
}
}

for(i=0;i<3;i++)
{
for(j=i+1;j<3;j++)
{
if(a[0][i]>a[0][j])
{
temp =a[0][i];
a[0][i] = a[0][j];
a[0][j] = temp;
}
}
}

for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
printf("%d\t",a[i][j]);
}
printf("\n");
}
}
