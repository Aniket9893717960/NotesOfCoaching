Q1) Write a Program to Print How many prime numbers between 1 to 1000.
                                          With Function
                                         ------------------
#include <stdio.h>
int prime(int n);

void main()
{
int n;
scanf("%d",&n);
for(int i=1;i<=n;i++){
if(prime(i)==1)
printf("%d",i);
}
}

int prime(int n){
int i;
if(n<=1)
return 0;
for(i=2;i<=(n/2);i++){
if(n%i==0){
return 0;
}
}
return 1;
}

Q2) How many numbers of '2'  in the four digit value ?

#include <stdio.h>
int func(int a);
void main()
{
int a=1222,count;
printf("Your Entered value is %d",a);
count = func(a);
printf("Total number of '2' in count is %d",count);
}

int func(int a){
static int sum=0,count;
if(a){
sum = a%10;
if(sum==2){
count++;
}
func(a/10);
}
return (count);
}
