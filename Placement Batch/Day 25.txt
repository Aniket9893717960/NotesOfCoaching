Q) write a program to calculate number of vowels and number of consonents?

#include <stdio.h>
void main()
{
char a[10];
int i,Vowel=0,consonent=0,wrong=0;
for(i=0;i<10;i++)
{
a[i]=getchar();
if(a[i]==13 || a[i]==10){
a[i]='\0';
break;
}
else if(a[i]<64 || (a[i]>90 && a[i]<97) || a[i]>122){
wrong++;
}
else {
switch(a[i]){
case 'A':Vowel++;break;
case 'E':Vowel++;break;
case 'I':Vowel++;break;
case 'O':Vowel++;break;
case 'U':Vowel++;break;
case 'a':Vowel++;break;
case 'e':Vowel++;break;
case 'i':Vowel++;break;
case 'o':Vowel++;break;
case 'u':Vowel++;break;
default:consonent++;
}
}}
printf("Number of vowels:%d",Vowel);
printf("Number of consonent:%d",consonent);
printf("Number of NOt alphabet:%d",wrong);
}
