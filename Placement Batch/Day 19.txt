void main()
{
int a[]={10,100,90,80,70,60,50,40,30,20,10},temp,n=10;

Q) Use while loop of i to left, j to right and check value is found and try to print it's index?

#include <stdio.h>
void main()
{
int a[] ={50,40,60,20,70,100,10,30,80,90},j,i,n=10,temp,value=10;
temp = n/2;
for(i=0;i<temp;i++)
{
if(a[i]^value==0){
printf("Value is Found at %d\n",i);
break;
}
}
for(j=n-1;j>temp;j--){
if(a[j]^value==0){
printf("Value is Found at %d\n",j);
break;
}
}
if(i==temp && j==temp)
printf("Not Found");
}
