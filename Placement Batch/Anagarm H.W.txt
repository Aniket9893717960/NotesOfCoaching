int sortAlphabet(char *ch,int length){
int i,j;
char temp;
for(i=0;i<length;i++){
for(j=i+1;j<length;j++){
if(ch[i]>ch[j]){
temp = ch[i];
ch[i] = ch[j];
ch[j] = temp;
}
} }
return 0;
}
