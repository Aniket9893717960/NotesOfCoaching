Q) Write a Program to calculate number os one's in a Given String "101101"?
                                                   OUR Experiment
                                                   -------------------- 

#include <stdio.h>

void main()
{
char ch[15];
int i,length,count=0;
for(i=0;i<15;i++)
{
ch[i] = getchar();
if(ch[i]==10 || ch[i]==13){
ch[i]='\0';
break;
}
length++;
}
printf("Your given value is: %s\n",ch);
printf("Your Entered String length Is:%d\n",length);
for(i=0;i<length;i++)
{
if(ch[i]=='1')
{
count++;
}
}
printf("Total number of one's in %s is %d",ch,count);
}
