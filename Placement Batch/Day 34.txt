Q) write a program to find minimum value of element at lowest Complexity?

#include <stdio.h>
void main()
{
int min,a,i;
scanf("%d",&min);
for(i=1;i<5;i++){
scanf("%d",&a);
if(min>a)
min=a;
}
printf("\n%d",min);
}
