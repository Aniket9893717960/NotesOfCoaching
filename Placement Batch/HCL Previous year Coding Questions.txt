class Ts
{
String moveHash(String str,int n){
int i=0;
char ch;
StringBuilder sb= new StringBuilder();
while(i<n){
ch = str.charAt(i);
if(ch=='#')
sb.append(ch);
i++;
 }
i=0;
while(i<n){
ch = str.charAt(i);
if(ch!='#')
sb.append(ch);
i++;
 }
return (sb.toString());
}

public static void main(String args[]){
Ts t1 = new Ts();
String str = args[0].toString();
String output = t1.moveHash(str,str.length());
System.out.print(output);
}
}
