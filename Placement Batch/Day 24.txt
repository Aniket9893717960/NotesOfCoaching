Q) Write a program to get a String From user & check  it is palindrome String or not?

#include <stdio.h>

void main()
{
char ch[10];
int length=0,i,j;
for(i=0;i<10;i++)
{
ch[i]=getchar();
if(ch[i]==13 || ch[i]==10){
ch[i]='\0';
break;
}
length++;
}

for(i=0,j=length-1;i<length/2 && j>length/2;i++,j--){
if((ch[i]^ch[j])!=0){
printf("Different");
break;
}
}
if(i==j)
printf("same\n");
}
