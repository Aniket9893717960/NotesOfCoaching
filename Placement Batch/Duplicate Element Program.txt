/*This will work only Two Different Duplicate Element*/
#include <stdio.h>
#define maxsize 8

void main(){
int a[maxsize]={5,2,1,2,4,3,4,0},b[maxsize]={0,0,0,0,0,0,0,0};
int i,n;

for(i=0;i<maxsize;i++)
{
 n = a[i];
 b[n-1] = b[n-1]+1;
 if(b[n-1]>1){
  n = a[i+1];
  b[n] = b[n]+1;
  i++; 
 }
}

for(i=0;i<maxsize;i++){
 if(b[i]>1)
 printf("Duplicate Element is %d",a[i]);
}
}
