#include <stdio.h>

int main(void) {
	int carry,i,j,x,lenArr;
	int N = 100;          
	int Arr[158] = {0};    
	
	Arr[0] = 1;
	lenArr = 1; // Number of elements in Arr
	
	carry = 0;
	for(i=1;i<=N;i++) {
		for(j=0;j<lenArr;j++) {
			x = Arr[j]*i + carry;
			Arr[j] = x%10;
			carry = x/10;
		}
		while(carry>0) {
			Arr[lenArr] = carry%10;
			carry = carry/10;
			lenArr++;
		}
	}
	for(i=lenArr-1;i>=0;i--)
		printf("%d",Arr[i]);
	
	return 0;
}
